def self.delete(url)
  uri = URI.parse(url)
  use_ssl = (uri.scheme == 'https')
  http = Net::HTTP.start(uri.host, uri.port, use_ssl: use_ssl)
  http.send_request('DELETE', uri.request_uri)
end

def self.get(url)
  uri = URI.parse(url)
  use_ssl = (uri.scheme == 'https')
  http = Net::HTTP.start(uri.host, uri.port, use_ssl: use_ssl)
  resp = http.send_request('GET', uri.request_uri)
  resp.body
end
